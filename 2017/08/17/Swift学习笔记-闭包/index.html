            runningTotal += amount
            return runningTotal
        }
        return incrementer

     // makeIncrementer(forIncrement: 10) 得到的是incrementer（）这个闭包
    let incrementByTen = makeIncrementer(forIncrement: 10)
    let incrementBySix = makeIncrementer(forIncrement: 6)
    // 闭包是引用类型
    let alsoIncrementByTen = incrementByTen  // 如果你将闭包赋值给了两个不同的常量或者变量，两个值都会指向同一个闭包本身

    // 如果把闭包 makeIncrementer(forIncrement: 10) 赋值给某个类的实例的属性，并且这个闭包访问了这个实例，那么将会造成循环引用 实例持有了闭包，闭包又引用了实例，死循环
    print(incrementByTen()) // 打印为10
